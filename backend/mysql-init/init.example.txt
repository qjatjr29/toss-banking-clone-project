-- 스키마 생성 (서비스별 논리 격리)
CREATE DATABASE IF NOT EXISTS toss_account CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATE DATABASE IF NOT EXISTS toss_transfer CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 서비스별 전용 유저 (서비스 간 DB 접근 차단)
CREATE USER IF NOT EXISTS '{ENV_ACCOUNT_USER}'@'%' IDENTIFIED BY '{ENV_ACCOUNT_PASSWORD}';
CREATE USER IF NOT EXISTS '{ENV_TRANSFER_USER}'@'%' IDENTIFIED BY '{ENV_TRANSFER_PASSWORD}';

GRANT ALL PRIVILEGES ON toss_account.* TO '{ENV_ACCOUNT_USER}'@'%';
GRANT ALL PRIVILEGES ON toss_transfer.* TO '{ENV_ACCOUNT_USER'@'%';
FLUSH PRIVILEGES;
